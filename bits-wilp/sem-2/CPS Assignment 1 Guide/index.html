    Gate Declarations
    id_t list[N+1];
int[0,N] len;

void enqueue(id_t element)
{
        list[len++] = element;
}

void reorganize_queue() {
    int i;

    int elementIndex = -1;
    for (i = 0; i < len; i++) {
        if (list[i] == PRIORITY) {
            elementIndex = i;
        }
    }

    if (elementIndex != -1) {
        for (i = elementIndex; i > 0; i--) {
            list[i] = list[i - 1];
        }
        list[0] = PRIORITY;
    }
}

void dequeue()
{
        int i = 0;
        len -= 1;
        while (i < len)
        {
                list[i] = list[i + 1];
                i++;
        }
        list[i] = 0;
        reorganize_queue();
}

id_t front()
{
   return list[0];
}

id_t tail()
{
   return list[len - 1];
}
